\project
	{Mitigating Featural Biases in Neural Nets}{}
	{
	    \textit{Advisors:  \href{https://sites.google.com/site/pshenoyuw/}{Dr. Pradeep Shenoy}, \href{https://praneethnetrapalli.org/}{Dr. Praneeth Netrapalli}}
	}
	{\begin{itemize}
	\setlength\itemsep{0.3em}
     \item Developed an interventional method for \textbf{addressing simplicity bias} in DNNs, called as \textit{feature sieve}. 
     \item Obtained upto \textbf{11.4\% relative gain} in accuracy over state-of-the-art methods on Imagenet-A. [\hyperref[C8]{ICML'23}]
     % \item Emprically and theoretically analysed the effect of our proposed methods  .
     \item Proposed an early readout mechanism to produce more robust models via distillation. [\hyperref[C7]{WACV'24}]
     \item Improved resnet18 student model by \textbf{5.2\% in worst-group accuracy} on CelebA.
     \end{itemize}}
